    # --verbosity 2                     # Moderate logging; shows trimming and splitting decisions (default: 1)

    # --min_trim_size 8                 # Do not trim adapters shorter than 8 bp (default: 8)

# End trimming settings: 
     
    # --end_threshold 70                # Minimum PID to trim an adapter at read ends; stricter than default (default: 75)

    # --end_size 150                    # Search first/last 150 bp of each read for end adapters (how much of the read to consider the "end"; rest is middle/internal) (default: 150)

    # --extra_end_trim 0                # No extra trimming beyond detected end adapter sequence (default: 0)
    

# Middle trimming settings:

    # --middle_threshold 80             # Minimum PID to detect internal adapters; very strict bc dont want to split reads by accident (default: 85)

    # --min_split_read_size 200         # Minimum size of read fragment to keep after splitting on an internal adapter (chimeric reads) (default: 1000)
    
    # --extra_middle_trim_good_side 0   # No extra trimming beyond detected adapter sequence on the "good" side of a middle adapter (default: 0)

    # --extra_middle_trim_bad_side 0    # No extra trimming beyond detected adapter sequence on the "bad" side of a middle adapter (default: 0)
